#include <Wire.h>
#include <Adafruit_MLX90614.h>
#include "MAX30105.h"
#include "heartRate.h"

Adafruit_MLX90614 mlx;
MAX30105 particleSensor;

// ================= PINS =================
#define MQ2_PIN A0
#define LED_PIN D5
#define BUZZER_PIN D6

// ================= THRESHOLDS =================
#define GAS_THRESHOLD 1025   // Change after calibration
#define TEMP_THRESHOLD 38.0
#define HR_HIGH 120
#define HR_LOW 50

long lastBeat = 0;
float bpm = 0;

void setup()
{
  Serial.begin(115200);

  pinMode(LED_PIN, OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);

  digitalWrite(LED_PIN, LOW);
  digitalWrite(BUZZER_PIN, LOW);

  // I2C Bus
  Wire.begin(D2, D1);

  // MLX90614
  if (!mlx.begin())
  {
    Serial.println("MLX90614 NOT FOUND!");
  }
  else
  {
    Serial.println("MLX90614 OK");
  }

  // MAX30102
  if (!particleSensor.begin(Wire, I2C_SPEED_STANDARD))
  {
    Serial.println("MAX30102 NOT FOUND!");
  }
  else
  {
    particleSensor.setup();
    particleSensor.setPulseAmplitudeRed(0x0A);
    particleSensor.setPulseAmplitudeGreen(0);

    Serial.println("MAX30102 OK");
  }

  Serial.println("SYSTEM STARTED");
}

void loop()
{
  bool danger = false;

  // ================= GAS SENSOR =================
  int gasValue = analogRead(MQ2_PIN);

  Serial.print("Gas Value: ");
  Serial.println(gasValue);

  if (gasValue > GAS_THRESHOLD)
  {
    danger = true;
    Serial.println("GAS ALERT!");
  }

  // ================= TEMPERATURE =================
  float ambient = mlx.readAmbientTempC();
  float object = mlx.readObjectTempC();

  Serial.print("Ambient Temp: ");
  Serial.print(ambient);
  Serial.println(" C");

  Serial.print("Body Temp: ");
  Serial.print(object);
  Serial.println(" C");

  if (object > TEMP_THRESHOLD)
  {
    danger = true;
    Serial.println("HIGH TEMPERATURE ALERT!");
  }

  // ================= HEART RATE =================
  long irValue = particleSensor.getIR();

  Serial.print("IR Value: ");
  Serial.println(irValue);

  if (irValue > 10000)
  {
    if (checkForBeat(irValue))
    {
      long delta = millis() - lastBeat;
      lastBeat = millis();

      bpm = 60.0 / (delta / 1000.0);

      Serial.print("Heart Rate: ");
      Serial.print(bpm);
      Serial.println(" BPM");

      if (bpm > HR_HIGH || bpm < HR_LOW)
      {
        danger = true;
        Serial.println("HEART RATE ALERT!");
      }
    }
  }
  else
  {
    Serial.println("Place Finger On Sensor");
  }

  // ================= ALERTS =================
  if (danger)
  {
    digitalWrite(LED_PIN, HIGH);
    digitalWrite(BUZZER_PIN, HIGH);
  }
  else
  {
    digitalWrite(LED_PIN, LOW);
    digitalWrite(BUZZER_PIN, LOW);
  }

  Serial.println("----------------------------");

  delay(1000);
}
